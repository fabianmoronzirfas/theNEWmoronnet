var doc  = apdocuments.add()
with(doc.documentPreferences){
	    pageWidth  = 160;
	    pageHeight = 160;
	    };
var p = doc.pages[0];
var ovs = new Array();
for(var i = 0; i &lt; 20 ; i++){
	    ovs[i] = ovals.add({geometricBounds:[10+i*3,10+i*3,150-i*3,150-i*3]});
	    ovs[i].strokeWeight = 0;
	    ovs[i].textPaths.add();
	    if(i &gt; 0){        
			ovs[i].textPaths[0].previousTextFrame = ovs[i-1].textPaths[0];
		        }
		    }
ovs[0].textPaths[0].contents = TextFrameContents.placeholderText;
